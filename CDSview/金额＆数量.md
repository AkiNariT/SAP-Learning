# 有金额和数量时一定要加！！！

## 金额使用
  ```js
@AbapCatalog.sqlViewName: 'ZVSDAMT01'
@AbapCatalog.compiler.compareFilter: true
@AbapCatalog.preserveKey: true
@AccessControl.authorizationCheck: #NOT_REQUIRED
@EndUserText.label: 'Sales Order Amount Example'
@Metadata.ignorePropagatedAnnotations: true
define view ZI_SD_AMOUNT_EXAMPLE
  as select from vbap
{
  key vbeln as SalesOrder,
  key posnr as SalesOrderItem,

      matnr as Material,

      @Semantics.amount.currencyCode: 'Currency'  //定义：这个字段是金额字段
      netwr as NetAmount, //NetAmount = 金额字段

      @Semantics.currencyCode: true  //定义：这个字段是货币码字段
      waerk as Currency //Currency = 通货字段
}
  ```

## 数量使用
  ```js
@AbapCatalog.sqlViewName: 'ZVSDQTY01'
@AbapCatalog.compiler.compareFilter: true
@AbapCatalog.preserveKey: true
@AccessControl.authorizationCheck: #NOT_REQUIRED
@EndUserText.label: 'Sales Order Quantity Example'
@Metadata.ignorePropagatedAnnotations: true
define view ZI_SD_QUANTITY_EXAMPLE
  as select from vbap
{
  key vbeln as SalesOrder,
  key posnr as SalesOrderItem,

      matnr as Material,

      @Semantics.quantity.unitOfMeasure: 'SalesUnit' //引用单位字段
      kwmeng as OrderQuantity, //OrderQuantity = 数量

      @Semantics.unitOfMeasure: true //声明单位字段
      vrkme as SalesUnit //SalesUnit = 单位
}

  ```
